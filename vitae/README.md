def compare_hobbies(hobbies_list: list) -> list:
    """ Returns similar hobbies. """
    my_hobbies = ['reading', 'dancing', 'travelling', 'singing']
    common = set()
    for i in hobbies_list:
        for j in my_hobbies:
            if i == j:
                common.add(j)
    return common
